--GunTeleport
function GunTp(Gunname)
	for _,GunReciver in pairs(workspace.AREA51:GetDescendants()) do 
		if GunReciver.name == "PUT THE WEAPON IN THIS BRICK" 
			and GunReciver.Parent.Parent.name == Gunname then 
			Humanoid.CFrame = GunReciver.CFrame * CFrame.new(Vector3.new(1,4,1)) 
		end end end  

--ItemTeleport
function ItemTp(RoomName)
	for _,ItemReciver in pairs(workspace.AREA51:GetDescendants()) do 
		if ItemReciver.name == "PUT THIS IN THE WEAPON" 
			and ItemReciver.Parent.Parent.name == RoomName then 
			Humanoid.CFrame = ItemReciver.Parent.CFrame * CFrame.new(Vector3.new(1,4,1)) 
		end end end
